function HPIA_V70_showClaspGithubStatus() {
  const status = HPIA_V70_getClaspGithubStatus_();
  const message = HPIA_V70_formatStatusMessage_(status);

  try {
    SpreadsheetApp.getUi().alert(message);
  } catch (err) {
    Logger.log(JSON.stringify(status, null, 2));
  }

  return status;
}

function HPIA_V70_openBackupDriveFolder() {
  return HPIA_V70_openLinkOrLog_(
    'Dossier backups live Apps Script',
    'URL du dossier backup Drive - voir documentation Drive privée V70R1'
  );
}

function HPIA_V70_openGithubRepo() {

function HPIA_V70_openLinkOrLog_(title, url) {
  try {
    HPIA_V70_showLinkDialog_(title, url);
  } catch (err) {
    const status = {
      ok: true,
      mode: 'LINK_LOG_ONLY_UI_UNAVAILABLE',
      title: title,
      url: url,
      message: 'UI unavailable in this execution context. Open the URL manually.'
    };
    Logger.log(JSON.stringify(status, null, 2));
    return status;
  }

  return {
    ok: true,
    mode: 'LINK_DIALOG_OPENED',
    title: title,
    url: url
  };
}

function HPIA_V70_showLinkDialog_(title, url) {
  const safeTitle = String(title).replace(/[<>&"]/g, function (c) {
    return ({ '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;' })[c];
  });
  const safeUrl = String(url).replace(/[<>&"]/g, function (c) {
    return ({ '<': '&lt;', '>': '&gt;', '&': '&amp;', '"': '&quot;' })[c];
  });

  const html = HtmlService
    .createHtmlOutput(
      '<div style="font:14px Arial,sans-serif;padding:16px;line-height:1.45">' +
      '<h2 style="font-size:16px;margin:0 0 12px">' + safeTitle + '</h2>' +
      '<p><a href="' + safeUrl + '" target="_blank" rel="noopener noreferrer">Ouvrir le lien</a></p>' +
      '<p style="color:#555;word-break:break-all">' + safeUrl + '</p>' +
      '</div>'
    )
    .setWidth(460)
    .setHeight(190);

  SpreadsheetApp.getUi().showModalDialog(html, safeTitle);
}
